package co.edu.unisabana.caf.repository;

import co.edu.unisabana.caf.models.Reservas;
import co.edu.unisabana.caf.models.Usuario;

import java.util.ArrayList;
import java.util.List;

public class ReservasRepository {
    List<Reservas> BD = new ArrayList<>();
    boolean resultado = true;
    public List<Reservas> BD(){
        BD.add(new Reservas("Lunes","7:00 am", "juliamcupa@unisabana.edu.co"));
        BD.add(new Reservas("Martes","11:00 am", "juliamcupa@unisabana.edu.co"));
        BD.add(new Reservas("Miercoles","8:00 am", "juliamcupa@unisabana.edu.co"));
        return BD;
    }

    public boolean crear(Reservas reservas){
        BD.add(reservas);
        return true;
    }

    public boolean verificar(Reservas reserva) {
        BD.forEach(dato -> {
            if (dato.getDia().equals(reserva.getDia()) && dato.getHora().equals(reserva.getHora())) {
                resultado = false;
            }
        });
        return resultado;
    }
}
